=== form | OSVRSTA | parent=(root)
--- BTN: DOLE
    PUSH KEY CLEAR
    SELECT OSVRSTA
    IF.NOT.EOF()
    SKIP
    ENDIF
    THISFORM.GRD0.SETFOCUS
    THISFORM.REFRESH
    POP KEY
    ENDPROC
--- BTN: GORE
    PUSH KEY CLEAR
    SELECT OSVRSTA
    IF.NOT.BoF()
    SKIP-1
    ENDIF
    THISFORM.GRD0.SETFOCUS
    THISFORM.REFRESH
    POP KEY
    ENDPROC
--- BTN: ZADNJI
    PUSH KEY CLEAR
    SELECT OSVRSTA
    IF.NOT.EOF()
    GO BOTTOM
    ENDIF
    THISFORM.GRD0.SETFOCUS
    THISFORM.REFRESH
    POP KEY
    ENDPROC
--- BTN: PRVI
    PUSH KEY CLEAR
    SELECT OSVRSTA
    IF.NOT.BoF()
    GO TOP
    ENDIF
    THISFORM.GRD0.SETFOCUS
    THISFORM.REFRESH
    POP KEY
    ENDPROC
--- BTN: \<DODAJ
    PUSH KEY CLEAR
    SELECT OSVRSTA
    APPEND BLANK
    IF RECNO()=RECCOUNT().OR.RECCOUNT()<1
    REPLACE VRSTA WITH STR(RECNO(),4,0)
    ENDIF
    THISFORM.REFRESH
    POP KEY
    ENDPROC
--- BTN: \<IZLAZ
    RELEASE THISFORM
    ENDPROC
