--- BTN: \<UČITAJ PODATKE
    PUSH KEY CLEAR
    USE OSPODACI IN 0
    select osPODACI
    SELECT OSPODACI
    MEDAT0=EDAT0
    MEDAT1=EDAT1
    USE OS IN 0
    select os
    SET FILTER TO EMPTY(NACINOB)
    SELECT OSOA
    GO TOP
    DO WHILE.NOT.EOF()
    MAG=AG
    SELECT OS
    GO TOP
    MPOCETNO=0
    MNEOTPIS=0
    MNABAVKA=0
    MPRODAJA=0
    MAMORT2=0
    MSAD2=0
    DO WHILE.NOT.EOF()
    IF AG=MAG
    IF DATUM0=MEDAT0
    MPOCETNO=MPOCETNO+SAD02
    ENDIF
    IF DATUM0>MEDAT0
    MNABAVKA=MNABAVKA+NAB02
    ENDIF
    IF DATUM1<MEDAT1
    MPRODAJA=MPRODAJA+SAD02
    ENDIF
    MNEOTPIS=MNEOTPIS+SAD02
    MAMORT2=MAMORT2+AMORT2
    MSAD2=MSAD2+SAD2
    ENDIF
    SKIP
    ENDDO
    SELECT OSOA
    REPLACE POCETNO WITH MPOCETNO
    REPLACE NEOTPIS WITH MNEOTPIS
    REPLACE NABAVKA WITH MNABAVKA
    REPLACE PRODAJA WITH MPRODAJA
    REPLACE AMORT2 WITH MAMORT2
    REPLACE SAD2 WITH MSAD2
    SKIP
    ENDDO
    SELECT OS
    SET FILTER TO
    USE
    SELECT OSPODACI
    USE
    SELECT OSOA
    GO TOP
    THISFORM.Release
    POP KEY
    KEYBOARD '{ENTER}'
    ENDPROC
