--- DOLE
    PUSH KEY CLEAR
    SELECT OS
    IF.NOT.EOF()
    SKIP
    ENDIF
    THISFORM.GRD0.SETFOCUS
    THISFORM.REFRESH
    POP key
    ENDPROC
--- GORE
    PUSH KEY CLEAR
    SELECT OS
    IF.NOT.BoF()
    SKIP-1
    ENDIF
    THISFORM.GRD0.SETFOCUS
    THISFORM.REFRESH
    POP key
    ENDPROC
--- ZADNJI
    PUSH KEY CLEAR
    SELECT OS
    IF.NOT.EOF()
    GO BOTTOM
    ENDIF
    THISFORM.GRD0.SETFOCUS
    THISFORM.REFRESH
    POP key
    ENDPROC
--- PRVI
    PUSH KEY CLEAR
    SELECT OS
    IF.NOT.BoF()
    GO TOP
    ENDIF
    THISFORM.GRD0.SETFOCUS
    THISFORM.REFRESH
    POP key
    ENDPROC
--- \<DODAJ
    PUSH KEY CLEAR
    SELECT OS
    APPEND BLANK
    THISFORM.GRD0.SETFOCUS
    THISFORM.REFRESH
    pop key
    ENDPROC
--- IZLAZ
    RELEASE THISFORM
    ENDPROC
--- POD\<ACI 
    DO FORM OSPODACI
    thisform.REFRESH()
    ENDPROC
--- \<PRENOS
    PUSH KEY CLEAR
    DO FORM OSPRENOS
    THISFORM.Release
    POP key
    KEYBOARD '{ENTER}'
    ENDPROC
--- \<OBRAČUN
    PUSH KEY CLEAR
    DO OSOBRAC
    GO TOP
    THISFORM.REFRESH
    POP key
    ENDPROC

--- SORT
    MORDER=TAGNO()
    DO CASE MORDER
    CASE MORDER=0
    SET ORDER TO 1
    CASE MORDER=1
    SET ORDER TO 2
    CASE MORDER=2
    SET ORDER TO 3
    CASE MORDER=3
    SET ORDER TO 4
    CASE MORDER=4
    SET ORDER TO 5
    CASE MORDER=5
    SET ORDER TO 6
    CASE MORDER=6
    SET ORDER TO 7
    CASE MORDER=7
    SET ORDER TO 8
    OTHERWISE
    SET ORDER TO
    ENDCASE
    THISFORM.GRD0.COLUMN6.SetFocus
    THISFORM.GRD0.Refresh
    THISFORM.GRD0.SETFOCUS
    ENDPROC
--- POA OBRAZAC
    PUSH KEY CLEAR
    USE FIRMA IN 0
    REPORT FORM ospoa NOCONSOLE prev
    SELECT firma
    USE
    SELECT os
    POP KEY
    ENDPROC
--- EVIDENCIJA POA
    PUSH KEY CLEAR
    SET cent off
    USE firma IN 0
    SELECT os
    SET FILTER TO !EMPTY(nacinob)
    REPORT FORM osevidencija NOCONSOLE prev
    SELECT firma
    use
    SELECT os
    SET FILTER TO
    SET CENTURY on
    POP KEY
    ENDPROC
--- PREGL. PORESKA nova OS.

    ENDPROC
