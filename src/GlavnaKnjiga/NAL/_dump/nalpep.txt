       PACK

       DO idel WITH (kkockax3)
       
       SELECT NALP
       GO TOP
       COPY TO &KKOCKA FOR SIFRA<>SPACE(5)
       GO TOP
       DO WHILE.NOT.EOF()
          IF SIFRA=SPACE(5)            
             SCATTER NAME POLJA

             GATHER NAME POLJA
             SELECT NALP
          ENDIF

       INDEX ON KONTO TAG KONTO

       ZAP

       USE
       SELECT NALP0
       APPEND FROM &KKOCKA
       USE
       USE &KKOCKA3 IN 0 ALIAS KOCKA EXCLU
       USE AAAN IN 0 ORDER 1
       SELECT AAAN
       SELECT KOCKA
       GO TOP

          MKONTO=KONTO

          MDUG=DUG

          SELECT AAAN
          SEEK MKONTO
          IF FOUND()
             MREC=RECNO()
             MAN='ANAL'+ALLTRIM(STR(MREC,2,0))
