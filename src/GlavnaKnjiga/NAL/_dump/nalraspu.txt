  METHODS (literal FoxPro source):
    PROCEDURE Click
    THISFORM.RELEASE
    MMTR=THISFORM.TXTMTR.Value
    MDAT0=THISFORM.TXTDAT0.Value
    MDAT1=THISFORM.TXTDAT1.Value
    MBRNAL=THISFORM.TXTBrnal.VALUE
    
    USE NAL IN 0 ORDER 1
    SELECT NALRASP
    SET ORDER TO 
    GO TOP
    DO WHILE.NOT.EOF()
       MKONTO=KONTO
       MMTR=MTR   
       MDP=DP
       REPLACE DATDOK WITH MDAT1
       REPLACE BRNAL WITH MBRNAL
       SELECT NAL
       SEEK MKONTO
       IF FOUND()
          MDUG=0
          MPOT=0
          DO WHILE.NOT.EOF()
             IF KONTO<>MKONTO
                EXIT
             ENDIF
             IF DTOS(DATDOK)>=DTOS(MDAT0).AND.DTOS(DATDOK)<=DTOS(MDAT1)
                IF MTR=MMTR.OR.MMTR=0
                   MDUG=MDUG+DUG
                   MPOT=MPOT+POT
                ENDIF     
             ENDIF   
             SKIP
          ENDDO      
          SELECT NALRASP
          IF MDP='D'
             REPLACE DUG WITH MDUG-MPOT
          ELSE
             REPLACE POT WITH MPOT-MDUG
          ENDIF
       ENDIF   
       SELECT NALRASP
       SKIP
    ENDDO
    SELECT NAL
    USE
    SELECT NALRASP
    GO TOP
    DO WHILE.NOT.EOF()
       MUCESCE=UCESCE
       REPLACE DUG WITH ROUND(DUG*MUCESCE/100,2)
       REPLACE POT WITH ROUND(POT*MUCESCE/100,2)
       SKIP
    ENDDO   
    GO TOP
    NALRASP.GRD0.REFRESH
    THISFORM.RELEASE   
    ENDPROC
