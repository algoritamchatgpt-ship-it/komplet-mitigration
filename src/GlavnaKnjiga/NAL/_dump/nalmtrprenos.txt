=== commandbutton | Command1 | parent=NALMTRPRENOS
  PROPERTIES:
    Top =   7.20
    Left =  26.40
    Height =   4.00

    FontSize = 11
    Picture = ..\ikone\aprenospara.png
    Caption = "\<Prenos"
    TabIndex = 11
    Name = "Command1"
  METHODS (literal FoxPro source):
    PROCEDURE Click
    PUSH KEY CLEAR
    MDAT0=THISFORM.TXTDAT0.VALUE
    MDAT1=THISFORM.TXTDAT1.VALUE
    DO IDEL WITH (KKOCKAX)
    DO IDEL WITH (KKOCKAX2)
    DO IDEL WITH (KKOCKAX3)
    SELECT NALMTR 
    SET ORDER TO 2
    USE &KKONTO IN 0 ORDER 1 ALIAS KONTO
    USE NAL IN 0 
    SELECT NAL
    COPY TO &KKOCKA FOR (SUBSTR(KONTO,1,1)='5'.OR.SUBSTR(KONTO,1,1)='6').AND.(DATDOK>=MDAT0.OR.DATDOK<=MDAT1)
    USE
    USE &KKOCKA IN 0 ALIAS KOCKA EXCLU
    SELECT KOCKA
    INDEX ON BRNAL TAG BRNAL
    INDEX ON KONTO+BRNAL TAG KONNAL
    SET ORDER TO 1
    TOTAL ON BRNAL FIELDS DUG,POT TO &KKOCKA3
    SET ORDER TO 2
    TOTAL ON KONTO+BRNAL FIELDS DUG,POT TO &KKOCKA2
    ZAP
    APPEND FROM &KKOCKA2
    SET ORDER TO 1
    GO TOP
    USE &KKOCKA3 IN 0 EXCLUSIVE ALIAS KOCKA3
    SELECT KOCKA3
    GO TOP
    DO WHILE.NOT.EOF()
       MBRNAL=BRNAL
       SELECT NALMTR
       SEEK MBRNAL 
       IF.NOT.FOUND() 
          SELECT KOCKA
          SEEK MBRNAL
          IF FOUND()
             DO WHILE.NOT.EOF()
                IF BRNAL<>MBRNAL
                   EXIT
                ENDIF     
                MKONTO=KONTO
                MDUG=DUG
                MPOT=POT
                MOPIS=OPIS
                MDATDOK=DATDOK
                SELECT KONTO
                SEEK MKONTO
                IF FOUND()
                   MNAZIV=NAZIV
                ELSE
                   MNAZIV=' '   
                ENDIF  
                SELECT NALMTR
                APPEND BLANK
                REPLACE KONTO WITH MKONTO
                REPLACE DATDOK WITH MDATDOK
                REPLACE BRNAL WITH MBRNAL
                REPLACE DUG WITH MDUG
                REPLACE POT WITH MPOT
                REPLACE OPIS WITH MOPIS
                REPLACE NAZIV WITH MNAZIV
                SELECT KOCKA
                SKIP
             ENDDO   
          ENDIF
       ENDIF
       SELECT KOCKA3
       SKIP
    ENDDO   
    
    SELECT KOCKA
    USE
    SELECT KOCKA3
    USE
    SELECT KONTO
    USE
    SELECT NALMTR
    SORT ON DATDOK,KONTO TO &KKOCKA2
    ZAP
    APPEND FROM &KKOCKA2
    GO TOP
    DO WHILE.NOT.EOF()
       REPLACE SALDO WITH DUG-POT 
       DO NALMTRRAC
       SKIP
    ENDDO   
    
    THISFORM.RELEASE
    NALMTR.RELEASE
    KEYBOARD '{ENTER}'
    POP KEY
    ENDPROC
