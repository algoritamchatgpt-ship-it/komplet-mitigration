    *-----------CIRILICA--------
    IF ACIRILICA=1
       MSEL=SELECT()
       mkscx=mbazni+'\'+'kscx.dbf'
       SET EXACT ON
       MNAME=LOWER(ALLTRIM(THISFORM.NAME))
       USE &mkscx IN 0 ORDER 1 ALIAS kscx
       SELECT kscx
       SEEK mname
       IF FOUND()
          DO while.not.eof()
             IF LOWER(ALLTRIM(imetab))<>mname
                EXIT
             endif   
             IF UPPER(ALLTRIM(kontrola))<>ALLTRIM(UPPER('form'))
                mstring=ALLTRIM(imekontr)+'.'+npodatak
             ELSE
                mstring=npodatak         
             endif   
             THISFORM.&MSTRING
             SKIP
          enddo   
       endif
       USE
       SELECT(msel)
       SET EXACT off
    ENDIF
